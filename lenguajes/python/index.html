¿Qué es Python?

Introducción

Python es un lenguaje de programación de alto nivel, interpretado y de propósito general. Fue creado por Guido van Rossum y lanzado por primera vez en 1991. Su diseño enfatiza la legibilidad del código y una sintaxis simple, lo que lo hace ideal tanto para principiantes como para desarrolladores experimentados

Más Informarción

Características principales

  • Sintaxis clara y legible, similar al inglés.

  • Multiplataforma: funciona en Windows, macOS y Linux.

  • Multiparadigma: admite programación orientada a objetos, estructurada y funcional.

  • Gran comunidad y amplia documentación disponible.

  • Tiene una gran cantidad de librerías y módulos para diversas tareas (como IA, análisis de datos, desarrollo web, automatización, etc.)

Usos comunes

  • Desarrollo web (con frameworks como Django o Flask).

  • Ciencia de datos y análisis estadístico (con Pandas, NumPy, Matplotlib).

  • Inteligencia artificial y machine learning (con TensorFlow, PyTorch).

  • Automatización de tareas y scripting.

  • Aplicaciones de escritorio y videojuegos.

  • Educación: muy usado para enseñar programación básica.

Ventajas y Desventajas

Ventajas

  • Código más corto y limpio en comparación con otros lenguajes como Java o C++.

  • Fácil de aprender y usar.

  • Compatible con otros lenguajes mediante extensiones.

Desventajas

  • Menor velocidad de ejecución que lenguajes compilados como C/C++.

  • No tan adecuado para aplicaciones que requieren alto rendimiento en tiempo real (aunque hay soluciones como Cython o PyPy).

1. Funciones Avanzadas

def calcular(a, b, operacion="suma"):
    if operacion == "suma":
        return a + b
    elif operacion == "resta":
        return a - b
    elif operacion == "multiplicacion":
        return a * b

resultado = calcular(10, 5, "multiplicacion")
print(resultado)  # 50

# Funciones con *args
def sumar_todo(*numeros):
    total = 0
    for n in numeros:
        total += n
    return total

print(sumar_todo(1, 2, 3, 4, 5))  # 15

# Funciones con **kwargs
def crear_perfil(**datos):

2. Decoradores

def decorador_tiempo(func):
    import time
    def wrapper(*args, **kwargs):
        inicio = time.time()
        resultado = func(*args, **kwargs)
        fin = time.time()
        print(f"Tiempo: {fin - inicio:.4f} segundos")
        return resultado
    return wrapper

@decorador_tiempo
def calcular_numeros_primos(n):
    primos = []
    for num in range(2, n):
        es_primo = True
        for divisor in range(2, int(num**0.5) + 1):
            if num % divisor == 0:
                es_primo = False
                break
        if es_primo:
            primos.append(num)
    return primos

resultado = calcular_numeros_primos(100)

3. Generadores

def fibonacci(n):
    a, b = 0, 1
    for _ in range(n):
        yield a
        a, b = b, a + b

# Usar el generador
for num in fibonacci(10):
    print(num)

# O convertir a lista
numeros = list(fibonacci(10))

6. Context Managers

from contextlib import contextmanager

@contextmanager
def recurso_controlado():
    print("Adquiriendo recurso")
    try:
        yield "Recurso disponible"
    finally:
        print("Liberando recurso")

with recurso_controlado() as recurso:
    print(f"Usando: {recurso}")

7. Ejemplo Complejo: API REST

import requests
import json

def obtener_informacion_usuario(user_id):
    """Obtiene información de un usuario de JSONPlaceholder"""
    try:
        url = f"https://jsonplaceholder.typicode.com/users/{user_id}"
        response = requests.get(url)
        response.raise_for_status()
        
        datos = response.json()
        return {
            "id": datos["id"],
            "nombre": datos["name"],
            "email": datos["email"],
            "ciudad": datos["address"]["city"]
        }
    except requests.exceptions.RequestException as e:
        print(f"Error en la solicitud: {e}")
        return None

# Usar la función
usuario = obtener_informacion_usuario(1)
if usuario:
    print(json.dumps(usuario, indent=2, ensure_ascii=False))

Ejemplo 1: Validador de Contraseñas

import re

def validar_contraseña(contraseña):
    """Valida que la contraseña cumpla criterios de seguridad"""
    requisitos = {
        "longitud": len(contraseña) >= 8,
        "mayuscula": bool(re.search(r'[A-Z]', contraseña)),
        "minuscula": bool(re.search(r'[a-z]', contraseña)),
        "numero": bool(re.search(r'\d', contraseña)),
        "especial": bool(re.search(r'[!@#$%^&*]', contraseña))
    }
    
    return all(requisitos.values()), requisitos

# Usar
valida, criterios = validar_contraseña("Mi2Contraseña!")
print(f"¿Válida?: {valida}")
print(f"Criterios: {criterios}")

Ejemplo 2: Gestor de Tareas

class GestorTareas:
    def __init__(self):
        self.tareas = []
    
    def agregar(self, titulo, prioridad=1):
        self.tareas.append({
            "titulo": titulo,
            "prioridad": prioridad,
            "completada": False
        })
    
    def completar(self, indice):
        if 0 <= indice < len(self.tareas):
            self.tareas[indice]["completada"] = True
    
    def ordenar_por_prioridad(self):
        return sorted(self.tareas, 
                     key=lambda x: x["prioridad"], 
                     reverse=True)
    
    def mostrar_pendientes(self):
        pendientes = [t for t in self.tareas if not t["completada"]]
        for i, tarea in enumerate(pendientes):
            print(f"{i+1}. {tarea['titulo']} [Prioridad: {tarea['prioridad']}]")

# Usar
gestor = GestorTareas()
gestor.agregar("Estudiar Python", 3)
gestor.agregar("Hacer proyecto", 2)
gestor.agregar("Descansar", 1)
gestor.mostrar_pendientes()

Ejemplo 3: Analizador de Texto

from collections import Counter

def analizar_texto(texto):
    """Analiza y muestra estadísticas de un texto"""
    # Limpiar y separar
    palabras = texto.lower().split()
    
    # Estadísticas
    stats = {
        "total_palabras": len(palabras),
        "palabras_unicas": len(set(palabras)),
        "longitud_promedio": sum(len(p) for p in palabras) / len(palabras),
        "palabra_mas_frecuente": Counter(palabras).most_common(1)[0]
    }
    
    return stats

# Usar
texto = "Python es un lenguaje Python que Python muchos"
resultado = analizar_texto(texto)
for clave, valor in resultado.items():
    print(f"{clave}: {valor}")
